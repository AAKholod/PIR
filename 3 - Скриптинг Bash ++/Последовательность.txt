--- Установка apache2 ---
1) cd ~
2) sudo openssl req -new -x509 -days 1461 -nodes -out cert.pem -keyout cert.key -subj "/C=RU/ST=Msk/L=Msk/O=PIX/OU=IT Department/CN=pixmaster"
3) sudo apt-get install apache2
4) sudo a2enmod proxy_http proxy_html proxy_wstunnel ssl
5) sudo a2enmod rewrite
6) sudo systemctl restart apache2
7) cd /etc/apache2
8) sudo mkdir ssl
9) cd ssl
10) sudo cp ~/cert.pem /etc/apache2/ssl
11) sudo cp ~/cert.key /etc/apache2/ssl
12) sudo nano /etc/apache2/apache2.conf
      ServerName pixmaster

sudo nano /etc/apache2/conf-enabled/pixmaster.conf

<VirtualHost *:443>
  ServerName pixmaster
  
  #Замените {DOMAINNAME} на доменное имя вашего сервера, такое же, как прописано в SSL-сертификате
  
  DocumentRoot /var/pixmaster 
  # стандартная папка установки PIX Master
  
  SSLEngine on
  SSLCertificateFile ssl/cert.pem
  SSLCertificateKeyFile ssl/cert.key
  #SSLCertificateChainFile ssl/cert.ca-bundle
  ProxyPreserveHost On
  ProxyPass / http://127.0.0.1:5000/
  ProxyPassReverse / http://127.0.0.1:5000/
  RewriteEngine on
  RewriteCond %{HTTPS} off
  RewriteRule (.*) https://%{HTTP_HOST}%{REQUEST_URI} [R=301,L]
  RewriteCond %{HTTP:UPGRADE} ^WebSocket$ [NC]
  RewriteCond %{HTTP:CONNECTION} Upgrade$ [NC]
  RewriteRule /(.*) ws://127.0.0.1:5000/$1 [P]
  ErrorLog /var/log/apache2/pixmaster-error.log
  CustomLog /var/log/apache2/pixmaster-access.log common
</VirtualHost>
13) sudo apachectl configtest
14) sudo service apache2 restart
